4 This is a minimized NTP Client library for Arduino to retrieve UNIX Epoch UTC from NTP time servers.

7 ## Library features
8 
9 * Retrieve UNIX Epoch UTC time from network time servers

11 * Timeout handling

37 ## Example ESP8266 / ESP32
38 
39 ```c++
40 #if defined(ARDUINO_ARCH_ESP8266)
41 #include <ESP8266WiFi.h>
42 #elif defined(ARDUINO_ARCH_ESP32)
43 #include <WiFi.h>
44 #endif
45 
46 #include <ErriezNTPClient.h>
47 
48 // WiFi SSID and Password
49 #define WIFI_SSID ""
50 #define WIFI_PASSWORD ""
51 
52 // "pool.ntp.org", "time.nist.gov" or NTP server IP address
53 #define NTP_SERVER "pool.ntp.org"
54 
55 ErriezNTPClient ntp(NTP_SERVER);
56 
57 
58 void setup()
59 {
60  // Initialize serial
61  delay(500);
62  Serial.begin(115200);
63  Serial.println(F("\nErriez NTP client ESP8266 / ESP32 example"));
64 
65  // Initialize WiFi
66  Serial.print(F("Connecting to '"));
67  Serial.print(WIFI_SSID);
68  Serial.print(F("'"));
69 
70  // Connect to your WiFi router
71  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
72 
73  // Wait for connection
74  while (WiFi.status() != WL_CONNECTED) {
75  delay(500);
76  Serial.print(".");
77  }
78  Serial.println("OK");
79 }
80 
81 void loop()
82 {
83  time_t t;
84 
85  // Get epoch
86  t = ntp.getEpoch();
87 
88  // Print result
89  if (t > 0) {
90  Serial.print(F("Epoch: "));
91  Serial.println((uint32_t)t);
92  Serial.print(F("UTC: "));
93  Serial.println(ctime(&t));
94  } else {
95  Serial.println(F("Timeout"));
96  }
97 
98  delay(10000);
99 }
100 ```

103 ## Example AVR (ATMega328 / ATMega2560)
104 
105 ```c++
106 #include <Ethernet.h>
107 #include <ErriezNTPClient.h>
108 
109 // "pool.ntp.org", "time.nist.gov" or NTP server IP address
110 #define NTP_SERVER "pool.ntp.org"
111 
112 ErriezNTPClient ntp(NTP_SERVER);
113 
114 // Newer Ethernet shields have a MAC address printed on a sticker on the shield
115 uint8_t mac[] = {
116  0xDE, 0xAD, 0xBE, 0xEF, 0xFE, 0xED
117 };
118 
119 
120 void setup()
121 {
122  // Initialize serial
123  delay(500);
124  Serial.begin(115200);
125  Serial.println(F("\nErriez NTP client AVR example"));
126 
127  // Start Ethernet and UDP
128  if (!Ethernet.begin(mac)) {
129  Serial.println(F("Failed to configure Ethernet using DHCP"));
130 
131  // Check for Ethernet hardware present
132  if (Ethernet.hardwareStatus() == EthernetNoHardware) {
133  Serial.println(F("Ethernet shield was not found."));
134  } else if (Ethernet.linkStatus() == LinkOFF) {
135  Serial.println(F("Ethernet cable is not connected."));
136  }
137  }
138 }
139 
140 void loop()
141 {
142  time_t t;
143 
144  // Get epoch
145  t = ntp.getEpoch();
146 
147  // Print result
148  if (t > 0) {
149  Serial.print(F("Epoch: "));
150  Serial.println((uint32_t)t);
151 
152  // A UNIX offset is needed for AVR target
153  t -= UNIX_OFFSET;
154 
155  Serial.print(F("UTC: "));
156  Serial.println(ctime(&t));
157  } else {
158  Serial.println(F("Timeout"));
159  }
160 
161  delay(10000);
162 }
163 ```
